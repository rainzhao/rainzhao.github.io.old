# angular指令

1.一个简单的指令

（1）指令的四种使用方式 

    //a.html
    <hello></hello>
    <div hello></div>
    <div class="hello"></div>
    <!-- directive:hello -->

    //a.js
    var myModule = angular.module('helloDirective',[]);
    myModule.directive('hello',function(){
        return{
            restrict: 'ECAM',
            template: '
Hi angular!
',
            replace: true
        }
    });
（2）使用templateUrl加载指令这样加载可写更多的html代码 
myModule.directive('hello',function(){
        return{
            restrict: 'ECAM',
            templateUrl: 'templates/hello.html',
            replace: true
        }
    })
（3）使用$templateCache将模版缓存起来然后加载 
 
    //run方法等所有模块注入完成后执行一次
    myModule.run(function($templateCache){
        $templateCache.put("1.html","
hello template Cache
");
    })

    myModule.directive("hello",function($templateCache){
        return{
            restrict : 'EACM',
            template: $templateCache.get("1.html"),
            replace: true
        }
    });
### 2.directive&directive(指令之间的请求) 创建一个新的指令，假设第一个指令已经应用，在父级元素上你可以请求到它或者在同一级的元素上。如果你需要访问一些功能在这个主要的指令上，通过暴露controller指令可以访问到它（注意这里的controller不是控制器中的controller）： 
ui.directive('a', function(){
  return {
    controller: function(){
      this.data = {}
      this.changeData = function( ... ) { ... }
    },
    link: ($scope, $element, $attributes, controller) {
      controller.data = { ... }
    }
  }
})

### 3.directive&controller(指令与控制器之间的请求) 
//directive&controller.js
var myModule = angular.module("myModule",[]);
myModule.controller("MyCtrl",["$scope",function($scope){
        $scope.loadData = function(){
            console.log("数据加载中....");
        }
    }])
    .controller("MyCtrl2",["$scope",function($scope){
        $scope.loadData2 = function(){
            console.log("数据加载中....2222");
        }
    }]);
//在不同的controller中复用指令 
myModule.directive("loader",function(){
    return{
        restrict:'AE',
        template:"
滑动加载
",
        replace: true,
        link:function(scope,element,attr){
            element.bind("mouseenter",function(){
                scope.$apply(attr.howtoload);
            })
        }
    }
})

//index.html
    <div ng-controller="MyCtrl">
        <loader howToLoad="loadData()"></loader>
    </div>
    <div ng-controller="MyCtrl2">
        <loader howToLoad="loadData2()"></loader>
    </div>
